#include <iostream>
using namespace std;

// funcion para saber si un numero es primo
bool esPrimo(int n) {
    if (n < 2) return false; 
    for (int i = 2; i < n; i++) { 
        if (n % i == 0) return false; 
    }
    return true;
}

// funcion que calcula la suma de los primeros 20 primos
int sumaPrimos() {
    int suma = 0;
    int contador = 0;
    int numero = 2; // empiezo desde el 2 porque es el primer primo

    while (contador < 20) { // quiero solo 20 primos
        if (esPrimo(numero)) {
            suma += numero;   // sumo el primo
            contador++;       // cuento uno mas
        }
        numero++; // paso al siguiente numero
    }

    return suma;
}

int main() {
    int resultado = sumaPrimos();
    cout << "La suma de los primeros 20 numeros primos es: " << resultado << endl;
    return 0;
}
